120  public boolean CommitTransaction(TxnContext txn_context) {
121  long[] partition_bid = txn_context.partition_bid;
122  if (partition_bid != null) {
123  long commit_ts;
124  for (int i = 0; i < access_list_.access_count_; ++i) {
125  Access access_ptr = access_list_.GetAccess(i);
126  Content content_ref = access_ptr.access_record_.content_;
127  if (access_ptr.access_type_ == READ_WRITE) {
128  int p = key_to_partition(access_ptr.access_record_.record_.GetPrimaryKey());
129  commit_ts = partition_bid[p];
130  assert (commit_ts >= access_ptr.timestamp_);
131  content_ref.SetTimestamp(commit_ts);
132  } else if (access_ptr.access_type_ == INSERT_ONLY) {
133  int p = key_to_partition(access_ptr.access_record_.record_.GetPrimaryKey());
134  commit_ts = partition_bid[p];
135  assert (commit_ts >= access_ptr.timestamp_);
136  content_ref.SetTimestamp(commit_ts);
137  } else if (access_ptr.access_type_ == DELETE_ONLY) {
138  int p = (int) Math.floor(Integer.parseInt(access_ptr.access_record_.record_.GetPrimaryKey()) / (int) (Math.ceil(NUM_ACCOUNTS / (double) thread_count_)));
139  commit_ts = partition_bid[p];
140  assert (commit_ts >= access_ptr.timestamp_);
141  content_ref.SetTimestamp(commit_ts);
142  }
143  }
144  } else {
145  long commit_ts = txn_context.getBID();//This makes the execution appears to execute at one atomic time unit. //GenerateMonotoneTimestamp(curr_epoch, GlobalTimestamp.GetMonotoneTimestamp());
146  for (int i = 0; i < access_list_.access_count_; ++i) {
147  Access access_ptr = access_list_.GetAccess(i);
148  Content content_ref = access_ptr.access_record_.content_;
149  if (access_ptr.access_type_ == READ_WRITE) {
150  assert (commit_ts >= access_ptr.timestamp_);
151  content_ref.SetTimestamp(commit_ts);
152  } else if (access_ptr.access_type_ == INSERT_ONLY) {
153  assert (commit_ts >= access_ptr.timestamp_);
154  content_ref.SetTimestamp(commit_ts);
155  } else if (access_ptr.access_type_ == DELETE_ONLY) {
156  assert (commit_ts >= access_ptr.timestamp_);
157  content_ref.SetTimestamp(commit_ts);
158  }
159  }
160  }
161  // release locks.
162  for (int i = 0; i < access_list_.access_count_; ++i) {
163  Access access_ptr = access_list_.GetAccess(i);
164  access_ptr.access_record_.content_.UnlockPartitions();
165  }
166  assert (access_list_.access_count_ <= kMaxAccessNum);
167  access_list_.Clear();
168  return true;
169  }
