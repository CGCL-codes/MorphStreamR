115  private static BitSet getReservedAffinity0() {
116  String reservedAffinity = System.getProperty(AFFINITY_RESERVED);
117  if (BASE_AFFINITY != null && (reservedAffinity == null || reservedAffinity.trim().isEmpty())) {
118  BitSet reserverable = new BitSet(PROCESSORS);
119  reserverable.set(1, PROCESSORS, true);
120  reserverable.andNot(BASE_AFFINITY);
121  if (reserverable.isEmpty() && PROCESSORS > 1) {
122  LoggerFactory.getLogger(AffinityLock.class).info("No isolated CPUs found, so assuming CPUs 1 to {} available.", (PROCESSORS - 1));
123  // make all but first CPUs available
124  reserverable.set(1, PROCESSORS);
125  return reserverable;
126  }
127  return reserverable;
128  }
129 
130  reservedAffinity = reservedAffinity.trim();
131  long[] longs = new long[1 + (reservedAffinity.length() - 1) / 16];
132  int end = reservedAffinity.length();
133  for (int i = 0; i < longs.length; i++) {
134  int begin = Math.max(0, end - 16);
135  longs[i] = Long.parseLong(reservedAffinity.substring(begin, end), 16);
136  end = begin;
137  }
138  return BitSet.valueOf(longs);
139  }

229  public static AffinityLock acquireLock(String desc) {
230  if (desc == null)
231  return LOCK_INVENTORY.noLock();
232 
233  desc = desc.toLowerCase();
234  int cpuId;
235  if (desc.startsWith("last")) {
236  String last = desc.substring(4);
237  int lastN;
238  if (last.isEmpty())
239  lastN = 0;
240  else
241  try {
242  lastN = Integer.parseInt(last);
243  } catch (NumberFormatException e) {
244  throw new IllegalArgumentException("Cannot parse '" + desc + "'", e);
245  }
246  if (lastN > 0)
247  throw new IllegalArgumentException("Cannot parse '" + desc + "'");
248 
249  cpuId = PROCESSORS + lastN - 1;
250 
251  } else if (desc.startsWith("csv:")) {
252  String content = desc.substring(4);
253  int[] cpus = Arrays.asList(content.split(",")).stream()
254  .map(String::trim)
255  .mapToInt(Integer::parseInt).toArray();
256 
257  return acquireLock(cpus);
258 
259  } else if (desc.equals("none")) {
260  return LOCK_INVENTORY.noLock();
261 
262  } else if (desc.equals("any")) {
263  return acquireLock();
264 
265  } else {
266  try {
267  cpuId = Integer.parseInt(desc);
268  } catch (NumberFormatException e) {
269  throw new IllegalArgumentException("Cannot parse '" + desc + "'", e);
270  }
271  }
272  if (cpuId <= 0) {
273  System.err.println("Cannot allocate 0 or negative cpuIds '" + desc + "'");
274  return LOCK_INVENTORY.noLock();
275  }
276  return acquireLock(cpuId);
277  }

369  throw new IllegalStateException("cpu " + cpuId + " already bound to " + assignedThread);
