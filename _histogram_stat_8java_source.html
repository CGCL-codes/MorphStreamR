65  int mean = getMean();
66  int meanIndex = mean / binWidth;
67  int confCount = 0;
68  int confidenceLimit = (count * confidenceLimitPercent) / 100;
69  int binCount = 0;
70  Bin bin = binMap.get(meanIndex);
71  if (null != bin) {
72  confCount += bin.getCount();
73  ++binCount;
74  }
75  //starting for mean index extend to both sides to include other bins
76  int offset = 1;
77  for (; binCount < binMap.size(); ++offset) {
78  bin = binMap.get(meanIndex + offset);
79  if (null != bin) {

81  ++binCount;

83  bin = binMap.get(meanIndex - offset);

87  }

91  }
