28  public MultiStreamOutputContoller(

32  HashMap<String, streaminfo> output_streams = op.getOutput_streams();
33  this.PClist = PClist;
34  collections = new HashMap<>();
35  counter = new HashMap<>();//to fight against GC.
36  for (String streamId : output_streams.keySet()) {
37  PartitionController[] PartitionControllers = new PartitionController[getPartitionController(streamId).size()];
38  for (int i = 0; i < getPartitionController(streamId).size(); i++) {
39  PartitionControllers[i] = getPartitionController(streamId).iterator().next();
40  }
41  counter.put(streamId, 0);
42  collections.put(streamId, PartitionControllers);
43  }
44  }
