54  expired = new ArrayList<>(expiredEvents);

57  lock.unlock();

145  private List<Event<T>> scanEvents(boolean fullScan) {
146  //LOG.DEBUG("Scan events, eviction policy {}", evictionPolicy);
147  List<T> eventsToExpire = new ArrayList<>();
148  List<Event<T>> eventsToProcess = new ArrayList<>();
149  try {
150  lock.lock();
151  Iterator<Event<T>> it = queue.iterator();
152  while (it.hasNext()) {
153  Event<T> windowEvent = it.next();
154  EvictionPolicy.Action action = evictionPolicy.evict(windowEvent);
155  if (action == EXPIRE) {
156  eventsToExpire.add(windowEvent.get());
157  it.remove();
158  } else if (!fullScan || action == STOP) {
159  break;
160  } else if (action == PROCESS) {
161  eventsToProcess.add(windowEvent);
162  }
163  }
164  expiredEvents.addAll(eventsToExpire);
165  } finally {
166  lock.unlock();
167  }
168  eventsSinceLastExpiry.set(0);
169  //LOG.DEBUG("[{}] events expired from window.", eventsToExpire.size());
170  if (!eventsToExpire.isEmpty()) {
171  //LOG.DEBUG("invoking windowLifecycleListener.onExpiry");
172  windowLifecycleListener.onExpiry(eventsToExpire);
173  }
