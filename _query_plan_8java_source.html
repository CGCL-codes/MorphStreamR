192  tableNames.add(this.startTableName);
193  tableNames.addAll(this.joinTableNames);
194  int pass = 1;
195  // Pass 1: Iterate through all partition tables. For each partition table, find
196  // the lowest cost QueryOperator to access that table. Construct a mapping
197  // of each table name to its lowest cost operator.
198  Map<Set, QueryOperator> map = new HashMap<>();
199  for (String table : tableNames) {
200  QueryOperator minOp = this.minCostSingleAccess(table);
201  Set<String> key = new HashSet<>();
202  key.add(table);
203  map.put(key, minOp);
204  }
205  // Pass i: On each pass, use the results from the previous pass to find the
206  // lowest cost joins with each partition table. Repeat until all tables have
207  // been joined.
208  Map<Set, QueryOperator> pass1Map = map;
209  Map<Set, QueryOperator> prevMap;
210  while (pass++ < tableNames.size()) {
211  prevMap = map;
212  map = this.minCostJoins(prevMap, pass1Map);
213  }

215  // operators, and return an iterator on the final operator
