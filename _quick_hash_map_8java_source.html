390  // New storage allocation
391  int newValueLen = (int) (newCapacity * loadFactor);
392  if (keyValueTable != null)
393  keyValueTable = Arrays.copyOf(keyValueTable, (newValueLen << keyIndexShift) + 1);
394  else
395  keyValueTable = new Object[(newValueLen << keyIndexShift) + 1];
396  int[] newIndices = new int[newCapacity + newValueLen];
397  if (indexTable != null) {
398  int mask = AVAILABLE_BITS ^ (hashLen - 1);
399  int newMask = AVAILABLE_BITS ^ (newCapacity - 1);
400  for (int i = hashLen - 1; i >= 0; i--) {
401  int j = indexTable[i];
402  if ((j & CONTROL_BITS) == CONTROL_EMPTY) continue;
403  if ((j & CONTROL_BITS) == CONTROL_NEXT) {
404  int i2 = (i + 1) & (hashLen - 1);
405  int j2 = indexTable[i2];
406  int arrayIndex1 = j & (hashLen - 1);
407  int arrayIndex2 = j2 & (hashLen - 1);
408  int newHashIndex1 = i | (j & (newMask ^ mask));
409  int newHashIndex2 = i | (j2 & (newMask ^ mask));
410  if (newHashIndex1 == newHashIndex2) {
411  newIndices[newHashIndex1] =
412  arrayIndex1 | (j & newMask) | CONTROL_NEXT;
413  newIndices[(newHashIndex1 + 1) & (newCapacity - 1)] =
414  arrayIndex2 | (j2 & newMask); // | CONTROL_EMPTY;
415  } else {
416  newIndices[newHashIndex1] = arrayIndex1 | (j & newMask) | CONTROL_END;
417  newIndices[newHashIndex2] = arrayIndex2 | (j2 & newMask) | CONTROL_END;
418  }
419  } else { // CONTROL_OVERFLOW and CONTROL_END
420  int next1i = -1, next1v = 0, next1n = 0;
421  int next2i = -1, next2v = 0, next2n = 0;
422  while (true) {
423  int arrayIndex = j & (hashLen - 1);
424  int newHashIndex = i | (j & (newMask ^ mask));
425  if (newHashIndex == i) {
426  if (next1i >= 0) {
427  newIndices[next1i] = next1v | CONTROL_OVERFLOW;
428  next1i = newCapacity + (next1v & (newCapacity - 1));
429  next1n++;
430  } else next1i = newHashIndex;
431  next1v = arrayIndex | (j & newMask);
432  } else if (newHashIndex == i + hashLen) {
433  if (next2i >= 0) {
434  newIndices[next2i] = next2v | CONTROL_OVERFLOW;
435  next2i = newCapacity + (next2v & (newCapacity - 1));
436  next2n++;
437  } else next2i = newHashIndex;
438  next2v = arrayIndex | (j & newMask);
439  } else {
440  int newIndex = arrayIndex | (j & newMask);
441  int oldIndex = newIndices[newHashIndex];
442  if ((oldIndex & CONTROL_BITS) != CONTROL_EMPTY) {
443  newIndices[newCapacity + arrayIndex] = oldIndex;
444  newIndex |= CONTROL_OVERFLOW;
445  } else newIndex |= CONTROL_END;
446  newIndices[newHashIndex] = newIndex;
447  }
448  if ((j & CONTROL_BITS) == CONTROL_END) break;
449  j = indexTable[hashLen + arrayIndex];
450  }
451  if (next1i >= 0) {
452  if (next1n == 1 && i != hashLen - 1 &&
453  (next1v & (hashLen - 1)) != 0 &&
454  newIndices[i + 1] == 0) {
455  newIndices[i] ^= CONTROL_OVERFLOW ^ CONTROL_NEXT;
456  newIndices[i + 1] = next1v;
457  } else
458  newIndices[next1i] = next1v | CONTROL_END;
459  }
460  if (next2i >= 0) {
461  if (next2n == 1 && i != hashLen - 1 &&
462  (next2v & (hashLen - 1)) != 0 &&
463  newIndices[i + hashLen + 1] == 0) {
464  newIndices[i + hashLen] ^= CONTROL_OVERFLOW ^ CONTROL_NEXT;

466  } else

492  // Check arrays lazy allocation
493  if (indexTable == null)

696  case CONTROL_NEXT:

747  keyValueTable[(firstUnusedIndex << keyIndexShift) + 2] = tmp;

832  @SuppressWarnings("unchecked")
833  final V removeKey(Object key, int index) {
834  // Null special case
835  if (key == null) {
836  if (nullKeyPresent) {
837  nullKeyPresent = false;
838  size--;
839  if (this instanceof FastLinkedHashMap)
840  removeHook(NULL_INDEX);
841  if (keyIndexShift > 0) {
842  V oldValue = (V) keyValueTable[0];
843  keyValueTable[0] = null;
844  return oldValue;
845  } else return (V) DUMMY_VALUE;
846  } else return (V) NOT_FOUND;
847  }
848  // Lazy array allocation check
849  if (indexTable == null)
850  return (V) NOT_FOUND;
851  // Compute hash index
852  int hc = hash(key.hashCode());
853  int prev = -1;
854  int curr = hc & (hashLen - 1);
855  // Check if this hash bin is empty
856  int i = indexTable[curr];
857  if ((i & CONTROL_BITS) == CONTROL_EMPTY)
858  return (V) NOT_FOUND;
859  // Search
860  int mask = AVAILABLE_BITS ^ (hashLen - 1);
861  label:
862  while (true) {
863  int j = i & (hashLen - 1);
864  int k = hashLen + j;
865  if ((hc & mask) == (i & mask)) {
866  boolean found;
867  if (index == NO_INDEX) {
868  Object o = keyValueTable[(j << keyIndexShift) + 1];
869  found = key == o || key.equals(o);
870  } else {
871  found = j == index;
872  }
873  if (found) {
874  size--;
875  switch (i & CONTROL_BITS) {
876  case CONTROL_END:
877  if (prev >= 0) {
878  indexTable[prev] |= CONTROL_END; // (indexTable[prev] & AVAILABLE_BITS)
879  } else {
880  indexTable[curr] = 0;
881  }
882  break;
883  case CONTROL_OVERFLOW:
884  indexTable[curr] = indexTable[k];
885  break;
886  case CONTROL_NEXT:
887  int c2 = (curr + 1) & (hashLen - 1);
888  int i2 = indexTable[c2];
889  indexTable[curr] = i2 | CONTROL_END; // & AVAILABLE_BITS
890  indexTable[c2] = 0;
891  break;
892  default: // CONTROL_EMPTY
893  indexTable[prev] |= CONTROL_END; // (indexTable[prev] & AVAILABLE_BITS)
894  indexTable[curr] = 0;
895  break;
896  }
897  if (size == 0) {
898  firstUnusedIndex = 0;
899  firstDeletedIndex = -1;
900  } else if (j == firstUnusedIndex - 1) {
901  firstUnusedIndex = j;
902  } else {
903  indexTable[k] = firstDeletedIndex;
904  firstDeletedIndex = j;
905  }
906  Object oldValue = index != NO_INDEX ? null :
907  keyIndexShift == 0 ? DUMMY_VALUE :
908  keyValueTable[(j << keyIndexShift) + 2];
909  keyValueTable[(j << keyIndexShift) + 1] = null;
910  if (keyIndexShift > 0) {
911  keyValueTable[(j << keyIndexShift) + 2] = null;
912  }
913  modCount++;
914  if (this instanceof FastLinkedHashMap) {
915  removeHook(j);
916  }
917  // validate("Remove "+key+", "+index);
918  return (V) oldValue;
919  }
920  }
921  prev = curr;
922  switch (i & CONTROL_BITS) {
923  case CONTROL_END:
924  break label;
925  case CONTROL_OVERFLOW:
926  curr = k;
927  break;
928  case CONTROL_NEXT:
929  curr = (curr + 1) & (hashLen - 1);
930  break;
931  default:
932  break label;
933  }
934  i = indexTable[curr];
935  }
936  return (V) NOT_FOUND;
937  }
