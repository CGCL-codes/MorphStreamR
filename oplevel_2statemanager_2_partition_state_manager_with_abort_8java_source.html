146  private void onRollbackAndRedoTransition(Operation operation, OnRollbackAndRedoSignal signal) {
147  DependencyType dependencyType = signal.getType();
148  OperationStateType parentState = signal.getState();
149  if (parentState.equals(OperationStateType.ABORTED)) {
150  // update dependencies and then check to rollback
151  // EXECUTED -> READY, must, notify children rollback
152  // BLOCKED -> READY, possible, put to ready queue
153  // READY -> READY, must, do nothing
154  // ABORTED -> ABORTED, do nothing
155  if (operation.getOperationState().equals(OperationStateType.EXECUTED)) {
156  operation.stateTransition(OperationStateType.READY);
157  readyAction(operation);
158  executableTaskListener.onOPRollbacked(operation);
159  notifyChildrenRollbackAndRedo(operation, OperationStateType.READY, OperationStateType.EXECUTED);
160  } else if (operation.getOperationState().equals(OperationStateType.BLOCKED)) {
161  if (!signal.getPrevParentState().equals(OperationStateType.EXECUTED)) {
162  operation.updateDependencies(dependencyType, parentState);
163  }
164  if (operation.tryReady()) {
165  readyAction(operation);
166  }
167  }
168  } else if (parentState.equals(OperationStateType.BLOCKED) || parentState.equals(OperationStateType.READY)) {
169  // EXECUTED -> BLOCKED, must
170  // READY -> BLOCKED, must
171  // BLOCKED -> BLOCKED, must
172  // ABORTED -> ABORTED, do nothing
173  if (operation.getOperationState().equals(OperationStateType.ABORTED)) {
174  return; // do nothing for already aborted operations
175  }
176  operation.updateDependencies(dependencyType, parentState);
177  operation.stateTransition(OperationStateType.BLOCKED);
178  if (operation.getOperationState().equals(OperationStateType.EXECUTED)) {
179  executableTaskListener.onOPRollbacked(operation);
180  // notify its children to rollback, otherwise just rollback its own state
181  notifyChildrenRollbackAndRedo(operation, OperationStateType.BLOCKED, OperationStateType.EXECUTED);
182  }
183  } else {
184  throw new UnsupportedOperationException();
185  }
186  }
