70  public static void sched_setaffinity(final int pid, final BitSet affinity) {
71  final CLibrary lib = CLibrary.INSTANCE;
72  final cpu_set_t cpuset = new cpu_set_t();
73  final int size = version.isSameOrNewer(VERSION_2_6) ? cpu_set_t.SIZE_OF_CPU_SET_T : NativeLong.SIZE;
74  final long[] bits = affinity.toLongArray();
75  for (int i = 0; i < bits.length; i++) {
76  if (Platform.is64Bit()) {
77  cpuset.__bits[i].setValue(bits[i]);
78  } else {
79  cpuset.__bits[i * 2].setValue(bits[i] & 0xFFFFFFFFL);
80  cpuset.__bits[i * 2 + 1].setValue((bits[i] >>> 32) & 0xFFFFFFFFL);
81  }
82  }
83  try {
84  if (lib.sched_setaffinity(pid, size, cpuset) != 0) {
85  throw new IllegalStateException("sched_setaffinity(" + pid + ", " + size +
86  ", 0x" + Utilities.toHexString(affinity) + ") failed; errno=" + Native.getLastError());
87  }
88  } catch (LastErrorException e) {
89  throw new IllegalStateException("sched_setaffinity(" + pid + ", " + size +
90  ", 0x" + Utilities.toHexString(affinity) + ") failed; errno=" + e.getErrorCode(), e);

94  public static int sched_getcpu() {
95  final CLibrary lib = CLibrary.INSTANCE;
96  try {
97  final int ret = lib.sched_getcpu();
98  if (ret < 0) {
99  throw new IllegalStateException("sched_getcpu() failed; errno=" + Native.getLastError());
100  }
101  return ret;
102  } catch (LastErrorException e) {
103  throw new IllegalStateException("sched_getcpu() failed; errno=" + e.getErrorCode(), e);
104  } catch (UnsatisfiedLinkError ule) {
105  try {
106  final IntByReference cpu = new IntByReference();
107  final IntByReference node = new IntByReference();
108  final int ret = lib.syscall(318, cpu, node, null);
109  if (ret != 0) {
110  throw new IllegalStateException("getcpu() failed; errno=" + Native.getLastError());
111  }
112  return cpu.getValue();
113  } catch (LastErrorException lee) {
114  if (lee.getErrorCode() == 38 && Platform.is64Bit()) { // unknown call
115  final Pointer getcpuAddr = new Pointer((-10L << 20) + 1024L * 2L);
116  final Function getcpu = Function.getFunction(getcpuAddr, Function.C_CONVENTION);
117  final IntByReference cpu = new IntByReference();
118  if (getcpu.invokeInt(new Object[]{cpu, null, null}) < 0) {
119  throw new IllegalStateException("getcpu() failed; errno=" + Native.getLastError());
120 
121  } else {
122  return cpu.getValue();
123  }
124  } else {
125  throw new IllegalStateException("getcpu() failed; errno=" + lee.getErrorCode(), lee);
126  }
127  }
