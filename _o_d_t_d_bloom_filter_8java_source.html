153  public double estimateCount(String item, long time) {
154  double res = Double.MAX_VALUE;
155  for (int bucketIndex : getHashBuckets(item)) {
156  // update the counter with the smoothing coeficient
157  double value = getBucket(bucketIndex) * Math.pow(beta, time - getTimer(bucketIndex));
158  setBucket(bucketIndex, value);
159  // update the bucket timer
160  setTimer(bucketIndex, time);
161  if (value < res)
162  res = value;
163  }
