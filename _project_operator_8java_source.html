68  protected RecordSchema computeSchema() throws QueryPlanException {
69  // check to make sure that the source operator is giving us columns that we project
70  RecordSchema sourceSchema = this.getSource().getOutputSchema();
71  List<String> sourceColumnNames = new ArrayList<>(sourceSchema.getFieldNames());
72  List<DataBox> sourceColumnTypes = new ArrayList<>(sourceSchema.getFieldTypes());
73  List<DataBox> columnTypes = new ArrayList<>();
74  for (String columnName : this.columns) {
75  columnName = this.checkSchemaForColumn(sourceSchema, columnName);
76  int sourceColumnIndex = sourceColumnNames.indexOf(columnName);
77  columnTypes.add(sourceColumnTypes.get(sourceColumnIndex));
78  this.indices.add(sourceColumnIndex);
79  }
80  if (this.sumColumn != null) {
81  this.sumColumn = this.checkSchemaForColumn(sourceSchema, this.sumColumn);
82  this.sumColumnIndex = sourceColumnNames.indexOf(this.sumColumn);
83  if (!(sourceColumnTypes.get(this.sumColumnIndex) instanceof IntDataBox) &&
84  !(sourceColumnTypes.get(this.sumColumnIndex) instanceof FloatDataBox)) {
85  throw new QueryPlanException("Cannot compute sum over a non-integer column: " + this.sumColumn + ".");
86  }
87  }
88  if (this.averageColumn != null) {
89  this.averageColumn = this.checkSchemaForColumn(sourceSchema, this.averageColumn);
90  this.averageColumnIndex = sourceColumnNames.indexOf(this.averageColumn);
91  if (!(sourceColumnTypes.get(this.averageColumnIndex) instanceof IntDataBox) &&
92  !(sourceColumnTypes.get(this.sumColumnIndex) instanceof FloatDataBox)) {
93  throw new QueryPlanException("Cannot compute sum over a non-integer column: " + this.averageColumn + ".");
94  }
95  }
96  // make sure we addOperation the correct columns to the output schema if we have aggregates in the
97  // projection
98  if (this.hasAggregate) {
99  if (this.hasCount) {
100  this.columns.add("countAgg");
101  columnTypes.add(new IntDataBox());
102  }
103  if (this.sumColumn != null) {
104  this.columns.add("sumAgg");
105  if (sourceColumnTypes.get(this.sumColumnIndex) instanceof IntDataBox) {
106  columnTypes.add(new IntDataBox());
107  this.sumIsFloat = false;
108  } else {
109  columnTypes.add(new FloatDataBox());
110  this.sumIsFloat = true;
111  }
112  }
113  if (this.averageColumn != null) {
114  this.columns.add("averageAgg");
115  columnTypes.add(new FloatDataBox());
116  }
117  }
118  return new RecordSchema(this.columns, columnTypes);

215  public SchemaRecord next() {
216  if (this.hasNext()) {
217  if (ProjectOperator.this.hasAggregate) {
218  while (this.sourceIterator.hasNext()) {
219  SchemaRecord r = this.sourceIterator.next();
220  List<DataBox> recordValues = r.getValues();
221  // if the d_record is a MarkerRecord, that means we reached the end of a group... we reset
222  // the aggregates and addOperation the appropriate new d_record to the new Records
223  if (r == this.markerRecord) {
224  if (ProjectOperator.this.hasCount) {
225  int count = ProjectOperator.this.getAndResetCount();
226  this.baseValues.add(new IntDataBox(count));
227  }
228  if (ProjectOperator.this.sumColumnIndex != -1) {
229  double sum = ProjectOperator.this.getAndResetSum();
230  if (ProjectOperator.this.sumIsFloat) {
231  this.baseValues.add(new FloatDataBox((float) sum));
232  } else {
233  this.baseValues.add(new IntDataBox((int) sum));
234  }
235  }
236  if (ProjectOperator.this.averageColumnIndex != -1) {
237  double average = (float) ProjectOperator.this.getAndResetAverage();
238  this.baseValues.add(new FloatDataBox((float) average));
239  }
240  // d_record that we just saw a marker d_record
241  this.prevWasMarker = true;
242  return new SchemaRecord(this.baseValues);
243  } else {
244  // if the previous d_record was a marker (or for the first d_record) we have to get the relevant
245  // fields out of the d_record
246  if (this.prevWasMarker) {
247  this.baseValues = new ArrayList<>();
248  for (int index : ProjectOperator.this.indices) {
249  this.baseValues.add(recordValues.get(index));
250  }
251  this.prevWasMarker = false;
252  }
253  if (ProjectOperator.this.hasCount) {
254  ProjectOperator.this.addToCount();
255  }
256  if (ProjectOperator.this.sumColumnIndex != -1) {
257  ProjectOperator.this.addToSum(r);
258  }
259  if (ProjectOperator.this.averageColumnIndex != -1) {
260  ProjectOperator.this.addToAverage(r);
261  }
262  }
263  }
264  // at the very end, we need to make sure we addOperation all the aggregated records to the result
265  // either because there was no group by or to addOperation the last group we saw
266  if (ProjectOperator.this.hasCount) {
267  int count = ProjectOperator.this.getAndResetCount();
268  this.baseValues.add(new IntDataBox(count));
269  }
270  if (ProjectOperator.this.sumColumnIndex != -1) {
271  double sum = ProjectOperator.this.getAndResetSum();
272  if (ProjectOperator.this.sumIsFloat) {
273  this.baseValues.add(new FloatDataBox((float) sum));
274  } else {
275  this.baseValues.add(new IntDataBox((int) sum));
276  }
277  }
278  if (ProjectOperator.this.averageColumnIndex != -1) {
279  double average = ProjectOperator.this.getAndResetAverage();
280  this.baseValues.add(new FloatDataBox((float) average));
281  }
282  return new SchemaRecord(this.baseValues);
283  } else {
284  SchemaRecord r = this.sourceIterator.next();
285  List<DataBox> recordValues = r.getValues();
286  List<DataBox> newValues = new ArrayList<>();
287  // if there is a marker d_record (in the case we're projecting from a group by), we simply
288  // leave the marker records in
289  if (r == this.markerRecord) {
290  return markerRecord;
291  } else {
292  for (int index : ProjectOperator.this.indices) {
293  newValues.add(recordValues.get(index));
294  }
295  return new SchemaRecord(newValues);
296  }
297  }
298  }
299  throw new NoSuchElementException();
