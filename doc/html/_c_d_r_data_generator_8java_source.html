181  public static String imei() {
182  int len = 15;
183  int len_offset = 0;
184  int pos = 0;
185  int t = 0;
186  int sum = 0;
187  int final_digit = 0;
188  int[] str = new int[len];
189  //
190  // Fill in the first two values of the string based with the specified prefix.
191  // Reporting Body Identifier list: http://en.wikipedia.org/wiki/Reporting_Body_Identifier
192  //
193  String[] arr = RBI[(int) Math.floor(Math.random() * RBI.length)].split("");
194  str[0] = Integer.parseInt(arr[1]);
195  str[1] = Integer.parseInt(arr[2]);
196  pos = 2;
197  //
198  // Fill all the remaining numbers except for the last one with random values.
199  //
200  while (pos < len - 1) {
201  str[pos++] = (int) Math.floor(Math.random() * 10) % 10;
202  }
203  //
204  // Calculate the Luhn checksum of the values thus far.
205  //
206  len_offset = (len + 1) % 2;
207  for (pos = 0; pos < len - 1; pos++) {
208  if ((pos + len_offset) % 2 != 0) {
209  t = str[pos] * 2;
210  if (t > 9) {
211  t -= 9;
212  }
213  sum += t;
214  } else {
215  sum += str[pos];
216  }
217  }
218  //
219  // Choose the last digit so that it causes the entire string to pass the checksum.
220  //
221  final_digit = (10 - (sum % 10)) % 10;
222  str[len - 1] = final_digit;
223  StringBuilder imei = new StringBuilder();
224  for (pos = 0; pos < len; pos++)
225  imei.append(str[pos]);
226  // Output the IMEI value.
227  return imei.toString();//.substring(0, len);
228  }
