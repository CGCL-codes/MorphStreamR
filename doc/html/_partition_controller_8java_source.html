70  protected PartitionController(TopologyComponent operator,
71  TopologyComponent childOP, HashMap<Integer, ExecutionNode> downExecutor_list
72  , int batch_size, ExecutionNode executionNode, Logger log, boolean profile, Configuration conf) {
73  this.childOP = childOP;
74  this.downExecutor_list = downExecutor_list;
75  this.batch = batch_size;
76  this.executionNode = executionNode;
77  LOG = log;
78  //foreach output stream, there could be different executors to listen on.
79  int sumweight = 0;
80  for (ExecutionNode consumer : downExecutor_list.values()) {
81  sumweight += consumer.compressRatio;
82  }
83  for (Map.Entry<Integer, ExecutionNode> e : downExecutor_list.entrySet()) {
84  //assume equal partition by-default.
85  partition_ratio.put(e.getKey(), (e.getValue().compressRatio / (double) sumweight));
86  }
87  Set<Integer> setID = downExecutor_list.keySet();
88  targetTasks = setID.toArray(new Integer[setID.size()]);
89  updateExtendedTargetId();
90  downTaskSize = targetTasks.length;
91  final ExecutionNode first = operator.getExecutorList().get(0);
92  firt_executor_Id = first.getExecutorID();
93  if (executionNode == null) {//shared.
94 // //LOG.DEBUG("MPSC controller is used.");
95  collections = new Collections[operator.getExecutorList().size()];
96  context = new TopologyContext[operator.getExecutorList().size()];
97  for (ExecutionNode src : operator.getExecutorList()) {
98  collections[(src.getExecutorID() - firt_executor_Id)] = new Collections(src.getExecutorID(), downExecutor_list, batch_size);
99  }
100  if (enable_log)
101  LOG.trace("MPSC implementation -- Queue is shared among multiple executors of the same producer.");
102  controller = new MPSCController(downExecutor_list);
103  } else {
104  firt_executor_Id = executionNode.getExecutorID();
105  collections = new Collections[1];
106  context = new TopologyContext[1];
107  collections[0] = new Collections(firt_executor_Id, downExecutor_list, batch_size);
108 
109  if (enable_log) LOG.trace("SPSC implementation -- Queue is unique to each producer and consumer");
110  controller = new SPSCController(downExecutor_list);
111  }
112  PartitionController.profile = profile;
113  threashold = conf.getInt("queue_size") - 1;//leave one space for watermark filling!
114  }

620  Collections(int src_Id, HashMap<Integer, ExecutionNode> DownExecutor_list, int batch_size) {

623  buffers = new JumboTuple[DownExecutor_list.size()];
