266  // Lower-case 'int' versions - no autoboxing, very fast.
267  // 'i' is known positive.
268  public boolean add(final int i) {
269  // Check for out-of-range for the current size bit vector.
270  // If so we need to grow the bit vector.
271  if ((i >> 6) >= _bits.length)
272  return install_larger_new_bits(i). // Install larger pile-o-bits (duh)
273  help_copy().add(i); // Finally, addOperation to the new table
274  // Handle every 64th bit via using a nested array
275  NBSI nbsi = this; // The bit array being added into
276  int j = i; // The bit index being added
277  while ((j & 63) == 63) { // Bit 64? (low 6 bits are all set)
278  nbsi = nbsi._nbsi64; // Recurse
279  j = j >> 6; // Strip off low 6 bits (all set)
280  }
281  final long mask = mask(j);
282  long old;
283  do {
284  old = nbsi._bits[j >> 6]; // Read old bits
285  if (old < 0) // Not mutable?
286  // Not mutable: finish copy of word, and retry on copied word
287  return help_copy_impl(i).help_copy().add(i);
288  if ((old & mask) != 0) return false; // Bit is already set?
289  } while (!nbsi.CAS(j >> 6, old, old | mask));
290  _size.add(1);
291  return true;
292  }

341  private NBSI install_larger_new_bits(final int i) {
342  if (_new == null) {
343  // Grow by powers of 2, to avoid minor grow-by-1's.
344  // Note: must grow by exact powers-of-2 or the by-64-bit trick doesn't work right
345  int sz = (_bits.length << 6) << 1;
346  // CAS to install a new larger size. Did it work? Did it fail? We
347  // don't know and don't care. Only One can be installed, so if
348  // another thread installed a too-small size, we can't help it - we
349  // must simply install our new larger size as a nested-resize table.
350  CAS_new(new NBSI(sz, _size, _non_blocking_set_int));
351  }
352  // Return self for 'fluid' programming style
353  return this;

356  // Help any top-level NBSI to copy until completed.
357  // Always return the _new version of *this* NBSI, in case we're nested.
358  private NBSI help_copy() {
359  // Pick some words to help with - but only help copy the top-level NBSI.
360  // Nested NBSI waits until the top is done before we start helping.
361  NBSI top_nbsi = _non_blocking_set_int._nbsi;
362  final int HELP = 8; // Tuning number: how much copy pain are we willing to inflict?
363  // We "help" by forcing individual bit indices to copy. However, bits
364  // come in lumps of 64 per word, so we just advance the bit counter by 64's.
365  int idx = top_nbsi._copyIdx.getAndAdd(64 * HELP);
366  for (int i = 0; i < HELP; i++) {
367  int j = idx + i * 64;
368  j %= (top_nbsi._bits.length << 6); // Limit, wrap to array size; means we retry indices
369  top_nbsi.help_copy_impl(j);
370  top_nbsi.help_copy_impl(j + 63); // Also force the nested-by-64 bit
371  }
372  // Top level guy ready to promote?
373  // Note: WE may not be the top-level guy!
374  if (top_nbsi._copyDone.get() == top_nbsi._sum_bits_length)
375  // One shot CAS to promote - it may fail since we are racing; others
376  // may promote as well
377  if (_non_blocking_set_int.CAS_nbsi(top_nbsi, top_nbsi._new)) {
378  //System.out.println("Promote at top level to size "+(_non_blocking_set_int._nbsi._bits.length<<6));
379  }
380  // Return the new bitvector for 'fluid' programming style
381  return _new;
382  }

384  // Help copy this one word. State Machine.
385  // (1) If not "made immutable" in the old array, set the sign bit to make
386  // it immutable.
387  // (2) If non-zero in old array & zero in new, CAS new from 0 to copy-of-old
388  // (3) If non-zero in old array & non-zero in new, CAS old to zero
389  // (4) Zero in old, new is valid
390  // At this point, old should be immutable-zero & new has a copy of bits
391  private NBSI help_copy_impl(int i) {
392  // Handle every 64th bit via using a nested array
393  NBSI old = this; // The bit array being copied from
394  NBSI nnn = _new; // The bit array being copied to
395  if (nnn == null) return this; // Promoted already
396  int j = i; // The bit index being added
397  while ((j & 63) == 63) { // Bit 64? (low 6 bits are all set)
398  old = old._nbsi64; // Recurse
399  nnn = nnn._nbsi64; // Recurse
400  j = j >> 6; // Strip off low 6 bits (all set)
401  }
402  // Transit from state 1: word is not immutable yet
403  // Immutable is in bit 63, the sign bit.
404  long bits = old._bits[j >> 6];
405  while (bits >= 0) { // Still in state (1)?
406  long oldbits = bits;
407  bits |= mask(63); // Target state of bits: sign-bit means immutable
408  if (old.CAS(j >> 6, oldbits, bits)) {
409  if (oldbits == 0) _copyDone.addAndGet(1);
410  break; // Success - old array word is now immutable
411  }
412  bits = old._bits[j >> 6]; // Retry if CAS failed
413  }
414  // Transit from state 2: non-zero in old and zero in new
415  if (bits != mask(63)) { // Non-zero in old?
416  long new_bits = nnn._bits[j >> 6];
417  if (new_bits == 0) { // New array is still zero
418  new_bits = bits & ~mask(63); // Desired new value_list: a mutable copy of bits
419  // One-shot CAS attempt, no loop, from 0 to non-zero.
420  // If it fails, somebody else did the copy for us
421  if (!nnn.CAS(j >> 6, 0, new_bits))
422  new_bits = nnn._bits[j >> 6]; // Since it failed, get the new value_list
423  assert new_bits != 0;
424  }
425  // Transit from state 3: non-zero in old and non-zero in new
426  // One-shot CAS attempt, no loop, from non-zero to 0 (but immutable)
427  if (old.CAS(j >> 6, bits, mask(63)))
428  _copyDone.addAndGet(1); // One more word finished copying
429  }
430  // Now in state 4: zero (and immutable) in old
431  // Return the self bitvector for 'fluid' programming style
432  return this;
433  }
