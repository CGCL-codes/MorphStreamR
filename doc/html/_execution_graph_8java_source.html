310  private PartitionController partitionController_create(TopologyComponent srcOP, TopologyComponent childOP
311  , String streamId, HashMap<Integer, ExecutionNode> downExecutor_list, int batch, ExecutionNode executor, boolean common) {
312  Grouping g = srcOP.getGrouping_to_downstream(childOP.getId(), streamId);

357  private void build_streamController(int batch) {
358  if (!shared) {
359  for (ExecutionNode executor : executionNodeArrayList) {//Build sc for each executor who is not leaf mapping_node.
360  if (executor.isLeafNode()) {
361  continue;
362  }

365  for (String streamId : executor.operator.getOutput_streamsIds()) {
366  HashMap<String, PartitionController> PClist = init_pc(streamId, executor.operator, batch, executor, common);//unique PC
367  PCMaps.put(streamId, PClist);
368  }
369  OutputController sc = new MultiStreamOutputContoller((MultiStreamComponent) executor.operator, PCMaps);
370  executor.setController(sc);
371  }
372  } else {
373  for (TopologyComponent operator : topology.getRecords().values()) {
374  if (operator.isLeafNode()) {
375  continue;
376  }
377  // streamId, DownOpId, PC
378  HashMap<String, HashMap<String, PartitionController>> PCMaps = new HashMap<>();
379  for (String streamId : operator.getOutput_streamsIds()) {
380  HashMap<String, PartitionController> PClist = init_pc(streamId, operator, batch, null, common);//shared PC
381  PCMaps.put(streamId, PClist);
382  }
383  OutputController sc = new MultiStreamOutputContoller((MultiStreamComponent) operator, PCMaps);
384  sc.setShared();
385  for (ExecutionNode executor : operator.getExecutorList()) {//double measure_end if they are refer to the same executor
