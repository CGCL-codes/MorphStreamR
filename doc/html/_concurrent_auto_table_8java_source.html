36  // Hash spreader
37  private static final int hash() {
38  int h = System.identityHashCode(Thread.currentThread());
39  // You would think that System.identityHashCode on the current thread
40  // would be a good hash fcn, but actually on SunOS 5.8 it is pretty lousy
41  // in the low bits.
42  h ^= (h >>> 20) ^ (h >>> 12); // Bit spreader, borrowed from Doug Lea
43  h ^= (h >>> 7) ^ (h >>> 4);
44  return h << 2; // Pad out cache lines. The goal is to avoid cache-line contention
45  }

183  // Only addOperation 'x' to some slot in table, hinted at by 'hash', if bits under
184  // the mask are all zero. The sum can overflow or 'x' can contain bits in
185  // the mask. Value is CAS'd so no counts are lost. The CAS is attempted
186  // ONCE.
187  public long add_if_mask(long x, long mask, int hash, ConcurrentAutoTable master) {
188  long[] t = _t;
189  int idx = hash & (t.length - 1);
190  // Peel loop; try once fast
191  long old = t[idx];
192  boolean ok = CAS(t, idx, old & ~mask, old + x);
193  if (_sum_cache != Long.MIN_VALUE)
194  _sum_cache = Long.MIN_VALUE; // Blow out cache
195  if (ok) return old; // Got it
196  if ((old & mask) != 0) return old; // Failed for bit-set under mask
197  // Try harder
198  int cnt = 0;
199  while (true) {
200  old = t[idx];
201  if ((old & mask) != 0) return old; // Failed for bit-set under mask
202  if (CAS(t, idx, old, old + x)) break; // Got it!
203  cnt++;
204  }
205  if (cnt < MAX_SPIN) return old; // Allowable spin loop count
206  if (t.length >= 1024 * 1024) return old; // too big already
207  // Too much contention; double array size in an effort to reduce contention
208  long r = _resizers;
209  int newbytes = (t.length << 1) << 3/*word to bytes*/;
210  while (!_resizerUpdater.compareAndSet(this, r, r + newbytes))
211  r = _resizers;
212  r += newbytes;
213  if (master._cat != this) return old; // Already doubled, don't bother
214  if ((r >> 17) != 0) { // Already too much allocation attempts?
215  // TODO - use a sync_ratio with timeout, so we'll wakeup as soon as the new
216  // table is ready, or after the timeout in any case. Annoyingly, this
217  // breaks the non-blocking property - so for now we just briefly sleep.
218  //synchronized( this ) { sync_ratio(8*megs); } // Timeout - we always wakeup
219  try {
220  Thread.sleep(r >> 17);
221  } catch (InterruptedException ignored) {
222  }
223  if (master._cat != this) return old;
224  }
225  CAT newcat = new CAT(this, t.length * 2, 0);
226  // Take 1 stab at updating the CAT with the new larger size. If this
227  // fails, we assume some other thread already expanded the CAT - so we
228  // do not need to retry until it succeeds.
229  master.CAS_cat(this, newcat);
230  return old;
231  }

233  // Return the current sum of all things in the table, stripping off mask
234  // before the addOperation. Writers can be updating the table furiously, so the
235  // sum is only locally accurate.
236  public long sum(long mask) {
237  long sum = _sum_cache;
238  if (sum != Long.MIN_VALUE) return sum;
239  sum = _next == null ? 0 : _next.sum(mask); // Recursively get cached sum
240  long[] t = _t;
241  for (int i = 0; i < t.length; i++)
242  sum += t[i] & (~mask);
243  _sum_cache = sum; // Cache includes recursive counts
244  return sum;
245  }
