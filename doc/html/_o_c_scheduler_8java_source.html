180  public void execute(ExecutionUnit operation, long mark_ID, boolean clean) {
181  if (operation.getOperationState().equals(MetaTypes.OperationStateType.ABORTED)) {
182  return; // return if the operation is already aborted
183  }
184  int success;
185  if (operation.accessType.equals(READ_WRITE_COND_READ)) {
186  success = operation.success[0];
187  Transfer_Fun(operation, mark_ID, clean);
188  if (operation.record_ref != null) {
189  operation.record_ref.setRecord(operation.d_record.content_.readPreValues(operation.bid));//read the resulting tuple.
190  }
191  if (operation.success[0] == success) {
192  operation.isFailed = true;
193  }
194  } else if (operation.accessType.equals(READ_WRITE_COND)) {
195  success = operation.success[0];
196  Transfer_Fun(operation, mark_ID, clean);
197  if (operation.success[0] == success) {
198  operation.isFailed = true;
199  }
200  } else if (operation.accessType.equals(READ_WRITE)) {
201  Depo_Fun(operation, mark_ID, clean);
202  } else if (operation.accessType.equals(READ_WRITE_COND_READN)) {
203  success = operation.success[0];
204  GrepSum_Fun(operation, mark_ID, clean);
205  if (operation.record_ref != null) {
206  operation.record_ref.setRecord(operation.d_record.content_.readPreValues(operation.bid));//read the resulting tuple.
207  }
208  if (operation.success[0] == success) {
209  operation.isFailed = true;
210  }
211  } else {
212  throw new UnsupportedOperationException();
213  }

257  public boolean executeWithBusyWait(Context context, SchedulingUnit operationChain, long mark_ID) {
258  MyList<ExecutionUnit> operation_chain_list = operationChain.getOperations();
259  for (ExecutionUnit operation : operation_chain_list) {
260  if (operation.getOperationState().equals(MetaTypes.OperationStateType.EXECUTED)
261  || operation.getOperationState().equals(MetaTypes.OperationStateType.ABORTED)
262  || operation.isFailed) continue;
263  if (isConflicted(context, operationChain, operation)) {
264  return false;
265  }
266  MeasureTools.BEGIN_SCHEDULE_USEFUL_TIME_MEASURE(context.thisThreadId);
267  execute(operation, mark_ID, false);
268  MeasureTools.END_SCHEDULE_USEFUL_TIME_MEASURE(context.thisThreadId);
269  if (!operation.isFailed && !operation.getOperationState().equals(MetaTypes.OperationStateType.ABORTED)) {
270  operation.stateTransition(MetaTypes.OperationStateType.EXECUTED);
271  } else {
272  checkTransactionAbort(operation, operationChain);
273  }
274  }
275  return true;
276  }
