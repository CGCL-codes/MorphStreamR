63  public void normalizeRows() {
64  //laplace correction
65  for (int r = 0; r < numRow; ++r) {
66  boolean gotZeroCount = false;
67  for (int c = 0; c < numCol && !gotZeroCount; ++c) {
68  gotZeroCount = table[r][c] == 0;
69  }
70  if (gotZeroCount) {
71  for (int c = 0; c < numCol; ++c) {
72  table[r][c] += 1;
73  }
74  }
75  }
76  //normalize
77  int rowSum = 0;
78  if (scale == 1) {
79  dTable = new double[numRow][numCol];
80  }
81  for (int r = 0; r < numRow; ++r) {
82  rowSum = getRowSum(r);
83  for (int c = 0; c < numCol; ++c) {
84  if (scale > 1) {
85  table[r][c] = (table[r][c] * scale) / rowSum;
86  } else {
87  dTable[r][c] = ((double) table[r][c]) / rowSum;
88  }
89  }
90  }
91  }
