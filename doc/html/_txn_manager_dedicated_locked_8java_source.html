51  long max_global_ts = max_rw_ts >> 32;
52  long max_local_ts = max_rw_ts & 0xFFFFFFFF;
53  assert (curr_epoch >= max_global_ts);
54  assert (curr_epoch >= this.local_epoch_);
55  // init.
56  if (curr_epoch > this.local_epoch_) {
57  this.local_epoch_ = curr_epoch;
58  this.local_ts_ = this.thread_id_;
59  }
60  assert (this.local_epoch_ == curr_epoch);
61  // compute commit timestamp.
62  if (curr_epoch == max_global_ts) {
63  if (this.local_ts_ <= max_local_ts) {
64  this.local_ts_ = (max_local_ts / thread_count_ + 1) * thread_count_ + thread_id_;
65  assert (this.local_ts_ > max_local_ts);
66  }
67  assert (this.local_ts_ > max_local_ts);
68  }

71  long commit_ts = (this.local_epoch_ << 32) | this.local_ts_;
72  assert (commit_ts >= max_rw_ts);
