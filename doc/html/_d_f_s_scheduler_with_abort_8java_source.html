173  protected void MarkOperationsToAbort(DFSLayeredTPGContextWithAbort context) {
174  boolean markAny = false;
175  ArrayList<DFSOperationChain> operationChains;
176  int curLevel;
177  for (Map.Entry<Integer, ArrayList<DFSOperationChain>> operationChainsEntry : context.allocatedLayeredOCBucket.entrySet()) {
178  operationChains = operationChainsEntry.getValue();
179  curLevel = operationChainsEntry.getKey();
180  for (DFSOperationChain operationChain : operationChains) {
181  for (DFSOperation operation : operationChain.getOperations()) {
182  markAny |= _MarkOperationsToAbort(context, operation);
183  }
184  }
185  if (markAny && context.rollbackLevel == -1) { // current layer contains operations to abort, try to abort this layer.
186  context.rollbackLevel = curLevel;
187  }
188  }
189  if (context.rollbackLevel == -1 || context.rollbackLevel > context.currentLevel) { // the thread does not contain aborted operations
190  context.rollbackLevel = context.currentLevel;
191  }
192  context.isRollbacked = true;
193  if (enable_log) LOG.debug("++++++ rollback at level: " + context.thisThreadId + " | " + context.rollbackLevel);
194  }

203  private boolean _MarkOperationsToAbort(DFSLayeredTPGContextWithAbort context, DFSOperation operation) {
