74  public static BloomSpecification computeBucketsAndK(double maxFalsePosProb) {
75  // Handle the trivial cases
76  if (maxFalsePosProb >= probs[minBuckets][minK]) {
77  return new BloomSpecification(2, optKPerBuckets[2]);
78  }
79  if (maxFalsePosProb < probs[maxBuckets][maxK]) {
80  return new BloomSpecification(maxK, maxBuckets);
81  }
82  // First find the minimal required number of buckets:
83  int bucketsPerElement = 2;
84  int K = optKPerBuckets[2];
85  while (probs[bucketsPerElement][K] > maxFalsePosProb) {
86  bucketsPerElement++;
87  K = optKPerBuckets[bucketsPerElement];
88  }
89  // Now that the number of buckets is sufficient, see if we can relax K
90  // without losing too much precision.
91  while (probs[bucketsPerElement][K - 1] <= maxFalsePosProb) {
92  K--;
93  }
94  return new BloomSpecification(K, bucketsPerElement);
