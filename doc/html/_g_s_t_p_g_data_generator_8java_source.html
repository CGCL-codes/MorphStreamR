188  int key;

190  int next = random.nextInt(100);
191  if (next < Ratio_of_Overlapped_Keys) { // randomly select a key from existing keyset.
192  if (!generatedKeys.isEmpty()) {
193  int counter = 0;

195  while (key / floor_interval != partitionId) {
196  key = generatedKeys.get(random.nextInt(generatedKeys.size()));
197  counter++;

202  }
203  }
