355  static boolean isNullVoltType(Object obj) {
356  boolean retval = false;
357  if (obj == null) {
358  retval = true;
359  } else if (obj == VoltType.NULL_TIMESTAMP ||
360  obj == VoltType.NULL_STRING ||
361  obj == VoltType.NULL_DECIMAL) {
362  retval = true;
363  } else {
364  switch (typeFromObject(obj)) {
365  case BOOLEAN:
366  retval = false; // HACK
367  break;
368  case TINYINT:
369  retval = (((Number) obj).byteValue() == NULL_TINYINT);
370  break;
371  case SMALLINT:
372  retval = (((Number) obj).shortValue() == NULL_SMALLINT);
373  break;
374  case INTEGER:
375  retval = (((Number) obj).intValue() == NULL_INTEGER);
376  break;
377  case BIGINT:
378  retval = (((Number) obj).longValue() == NULL_BIGINT);
379  break;
380  case FLOAT:
381  retval = (((Number) obj).doubleValue() == NULL_FLOAT);
382  break;
383  case TIMESTAMP:
384  retval = (obj == VoltType.NULL_TIMESTAMP);
385  break;
386  case STRING:
387  retval = (obj == VoltType.NULL_STRING);
388  break;
389  case DECIMAL:
390  retval = (obj == VoltType.NULL_DECIMAL);
391  break;
392  default:
393  throw new VoltTypeException("Unsupported type: " +
394  typeFromObject(obj));
395  }
396  }
397  return retval;
398  }

525  // Really hacky cast overflow detection for primitive types
526  // Comparison to MIN_VALUEs are <= to avoid collisions with the NULL
527  // bit pattern
528  // Probably eventually want a generic wouldCastDiscardInfo() call or
529  // something
530  boolean wouldCastOverflow(Number value) {
531  boolean retval = false;
532  switch (this) {
533  case TINYINT:
534  if (value.longValue() <= Byte.MIN_VALUE ||
535  value.longValue() > Byte.MAX_VALUE) {
536  retval = true;
537  }
538  break;
539  case SMALLINT:
540  if (value.longValue() <= Short.MIN_VALUE ||
541  value.longValue() > Short.MAX_VALUE) {
542  retval = true;
543  }
544  break;
545  case INTEGER:
546  if (value.longValue() <= Integer.MIN_VALUE ||
547  value.longValue() > Integer.MAX_VALUE) {
548  retval = true;
549  }
550  break;
551  case BIGINT:
552  // overflow isn't detectable for Longs, just look for NULL value_list
553  // In practice, I believe that we should never get here in VoltTable
554  // since we check for NULL before checking for cast overflow
555  if (value.longValue() == NULL_BIGINT) {
556  retval = true;
557  }
558  break;
559  case FLOAT:
560  // this really should never occur, also, just look for NULL
561  // In practice, I believe that we should never get here in VoltTable
562  // since we check for NULL before checking for cast overflow
563  if (value.doubleValue() == NULL_FLOAT) {
564  retval = true;
565  }
566  break;
567  default:
568  throw new VoltTypeException("Unhandled cast overflow case, " +
569  "casting to: " + toString());
570  }
571  return retval;
572  }
