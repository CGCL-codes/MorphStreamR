149  public static int[] createHashes(byte[] data, int hashes) {
150  int[] result = new int[hashes];
151  int k = 0;
152  byte salt = 0;
153  while (k < hashes) {
154  byte[] digest;
155  synchronized (digestFunction) {
156  digestFunction.update(salt);
157  salt++;
158  digest = digestFunction.digest(data);
159  }
160  for (int i = 0; i < digest.length / 4 && k < hashes; i++) {
161  int h = 0;
162  for (int j = (i * 4); j < (i * 4) + 4; j++) {
163  h <<= 8;
164  h |= ((int) digest[j]) & 0xFF;
165  }
166  result[k] = h;
167  k++;
168  }
169  }
170  return result;
171  }
