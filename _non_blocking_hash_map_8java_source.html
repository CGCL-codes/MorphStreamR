156  // --- hash ----------------------------------------------------------------
157  // Helper function to spread lousy hashCodes
158  private static final int hash(final Object key) {
159  int h = key.hashCode(); // The real hashCode call
160  // Spread bits to regularize both segment and index locations,
161  // using variant of single-word Wang/Jenkins hash.
162  h += (h << 15) ^ 0xffffcd7d;
163  h ^= (h >>> 10);
164  h += (h << 3);
165  h ^= (h >>> 6);
166  h += (h << 2) + (h << 14);
167  return h ^ (h >>> 16);
168  }

207  // --- reprobe_limit -----------------------------------------------------
208  // Heuristic to decide if we have reprobed toooo many times. Running over
209  // the reprobe limit on a 'get' call acts as a 'miss'; on a 'put' call it
210  // can trigger a table resize. Several places must have exact agreement on
211  // what the reprobe_limit is, so we share it here.
212  private static final int reprobe_limit(int len) {
213  return REPROBE_LIMIT + (len >> 2);

216  // --- keyeq ---------------------------------------------------------------
217  // Check for key equality. Try direct pointer compare first, then see if
218  // the hashes are unequal (fast negative test) and finally do the full-on
219  // 'equals' v-call.
220  private static boolean keyeq(Object K, Object key, int[] hashes, int hash, int fullhash) {
221  return
222  K == key || // Either keys match exactly OR
223  // hash exists and matches? hash can be zero during the install of a
224  // new key/value_list pair.
225  ((hashes[hash] == 0 || hashes[hash] == fullhash) &&
226  // Do not call the users' "equals()" call with a Tombstone, as this can
227  // surprise poorly written "equals()" calls that throw exceptions
228  // instead of simply returning false.
229  K != TOMBSTONE && // Do not call users' equals call with a Tombstone
230  // Do the match the hard way - with the users' key being the loop-
231  // invariant "this" pointer. I could have flipped the order of
232  // operands (since equals is commutative), but I'm making mega-morphic
233  // v-calls in a reprobing loop and nailing down the 'this' argument
234  // gives both the JIT and the hardware a chance to prefetch the call target.
235  key.equals(K)); // Finally do the hard match
236  }

238  private static final Object get_impl(final NonBlockingHashMap topmap, final Object[] kvs, final Object key, final int fullhash) {
239  final int len = len(kvs); // Count of key/value_list pairs, reads kvs.length
240  final CHM chm = chm(kvs); // The CHM, for a volatile read below; reads slot 0 of kvs
241  final int[] hashes = hashes(kvs); // The memoized hashes; reads slot 1 of kvs
242  int idx = fullhash & (len - 1); // First key hash
243  // Main spin/reprobe loop, looking for a Key hit
244  int reprobe_cnt = 0;
245  while (true) {
246  // Probe table. Each read of 'val' probably misses in cache in a big
247  // table; hopefully the read of 'key' then hits in cache.
248  final Object K = key(kvs, idx); // Get key before volatile read, could be null
249  final Object V = val(kvs, idx); // Get value_list before volatile read, could be null or Tombstone or Prime
250  if (K == null) return null; // A clear miss
251  // We need a volatile-read here to preserve happens-before semantics on
252  // newly inserted Keys. If the Key body was written just before inserting
253  // into the table a Key-compare here might read the uninitalized Key body.
254  // Annoyingly this means we have to volatile-read before EACH key compare.
255  // .
256  // We also need a volatile-read between reading a newly inserted Value
257  // and returning the Value (so the user might end up reading the stale
258  // Value contents). Same problem as with keys - and the one volatile
259  // read covers both.
260  final Object[] newkvs = chm._newkvs; // VOLATILE READ before key compare
261  // Key-compare
262  if (keyeq(K, key, hashes, idx, fullhash)) {
263  // Key hit! Check for no table-copy-in-progress
264  if (!(V instanceof Prime)) // No copy?
265  return (V == TOMBSTONE) ? null : V; // Return the value_list
266  // Key hit - but slot is (possibly partially) copied to the new table.
267  // Finish the copy & retry in the new table.
268  return get_impl(topmap, chm.copy_slot_and_check(topmap, kvs, idx, key), key, fullhash); // Retry in the new table
269  }
270  // get and put must have the same key lookup logic! But only 'put'
271  // needs to force a table-resize for a too-long key-reprobe sequence.
272  // Check for too-many-reprobes on get - and flip to the new table.
273  if (++reprobe_cnt >= reprobe_limit(len) || // too many probes
274  key == TOMBSTONE) // found a TOMBSTONE key, means no more keys in this table
275  return newkvs == null ? null : get_impl(topmap, topmap.help_copy(newkvs), key, fullhash); // Retry in the new table
276  idx = (idx + 1) & (len - 1); // Reprobe by 1! (could now prefetch)
277  }
278  }

280  // --- putIfMatch ---------------------------------------------------------
281  // Put, Remove, PutIfAbsent, etc. Return the old value_list. If the returned
282  // value_list is equal to expVal (or expVal is NO_MATCH_OLD) then the put can be
283  // assumed to work (although might have been immediately overwritten). Only
284  // the path through copy_slot passes in an expected value_list of null, and
285  // putIfMatch only returns a null if passed in an expected null.
286  private static final Object putIfMatch(final NonBlockingHashMap topmap, final Object[] kvs, final Object key, final Object putval, final Object expVal) {
287  assert putval != null;
288  assert !(putval instanceof Prime);
289  assert !(expVal instanceof Prime);
290  final int fullhash = hash(key); // throws NullPointerException if key null
291  final int len = len(kvs); // Count of key/value_list pairs, reads kvs.length
292  final CHM chm = chm(kvs); // Reads kvs[0]
293  final int[] hashes = hashes(kvs); // Reads kvs[1], read before kvs[0]
294  int idx = fullhash & (len - 1);
295  // ---
296  // Key-Claim stanza: spin till we can claim a Key (or force a resizing).
297  int reprobe_cnt = 0;
298  Object K = null, V = null;
299  Object[] newkvs = null;
300  while (true) { // Spin till we get a Key slot
301  V = val(kvs, idx); // Get old value_list (before volatile read below!)
302  K = key(kvs, idx); // Get current key
303  if (K == null) { // Slot is free?
304  // Found an empty Key slot - which means this Key has never been in
305  // this table. No need to put a Tombstone - the Key is not here!
306  if (putval == TOMBSTONE) return putval; // Not-now & never-been in this table
307  // Claim the null key-slot
308  if (CAS_key(kvs, idx, null, key)) { // Claim slot for Key
309  chm._slots.add(1); // Raise key-slots-used count
310  hashes[idx] = fullhash; // Memoize fullhash
311  break; // Got it!
312  }
313  // CAS to claim the key-slot failed.
314  //
315  // This re-read of the Key points out an annoying short-coming of Java
316  // CAS. Most hardware CAS's report back the existing value_list - so that
317  // if you fail you have a *witness* - the value_list which caused the CAS
318  // to fail. The Java API turns this into a boolean destroying the
319  // witness. Re-reading does not recover the witness because another
320  // thread can write over the memory after the CAS. Hence we can be in
321  // the unfortunate situation of having a CAS fail *for cause* but
322  // having that cause removed by a later store. This turns a
323  // non-spurious-failure CAS (such as Azul has) into one that can
324  // apparently spuriously fail - and we avoid apparent spurious failure
325  // by not allowing Keys to ever change.
326  K = key(kvs, idx); // CAS failed, get updated value_list
327  assert K != null; // If keys[idx] is null, CAS shoulda worked
328  }
329  // Key slot was not null, there exists a Key here
330  // We need a volatile-read here to preserve happens-before semantics on
331  // newly inserted Keys. If the Key body was written just before inserting
332  // into the table a Key-compare here might read the uninitalized Key body.
333  // Annoyingly this means we have to volatile-read before EACH key compare.
334  newkvs = chm._newkvs; // VOLATILE READ before key compare
335  if (keyeq(K, key, hashes, idx, fullhash))
336  break; // Got it!
337  // get and put must have the same key lookup logic! Lest 'get' give
338  // up looking too soon.
339  //topmap._reprobes.addOperation(1);
340  if (++reprobe_cnt >= reprobe_limit(len) || // too many probes or
341  key == TOMBSTONE) { // found a TOMBSTONE key, means no more keys
342  // We simply must have a new table to do a 'put'. At this point a
343  // 'get' will also go to the new table (if any). We do not need
344  // to claim a key slot (indeed, we cannot find a free one to claim!).
345  newkvs = chm.resize(topmap, kvs);
346  if (expVal != null) topmap.help_copy(newkvs); // help along an existing copy
347  return putIfMatch(topmap, newkvs, key, putval, expVal);
348  }
349  idx = (idx + 1) & (len - 1); // Reprobe!
350  } // End of spinning till we get a Key slot
351  // ---
352  // Found the proper Key slot, now update the matching Value slot. We
353  // never put a null, so Value slots monotonically move from null to
354  // not-null (deleted Values use Tombstone). Thus if 'V' is null we
355  // fail this fast cutout and fall into the check for table-full.
356  if (putval == V) return V; // Fast cutout for no-change
357  // See if we want to move to a new table (to avoid high average re-probe
358  // counts). We only check on the set_executor_ready set of a Value from null to
359  // not-null (i.e., once per key-put). Of course we got a 'free' check
360  // of newkvs once per key-compare (not really free, but paid-for by the
361  // time we get here).
362  if (newkvs == null && // New table-copy already spotted?
363  // Once per fresh key-put check the hard way
364  ((V == null && chm.tableFull(reprobe_cnt, len)) ||
365  // Or we found a Prime, but the JMM allowed reordering such that we
366  // did not spot the new table (very rare race here: the writing
367  // thread did a CAS of _newkvs then a store of a Prime. This thread
368  // reads the Prime, then reads _newkvs - but the read of Prime was so
369  // delayed (or the read of _newkvs was so accelerated) that they
370  // swapped and we still read a null _newkvs. The resize call below
371  // will do a CAS on _newkvs forcing the read.
372  V instanceof Prime))
373  newkvs = chm.resize(topmap, kvs); // Force the new table copy to start
374  // See if we are moving to a new table.
375  // If so, copy our slot and retry in the new table.
376  if (newkvs != null)
377  return putIfMatch(topmap, chm.copy_slot_and_check(topmap, kvs, idx, expVal), key, putval, expVal);
378  // ---
379  // We are finally prepared to update the existing table
380  while (true) {
381  assert !(V instanceof Prime);
382  // Must match old, and we do not? Then bail out now. Note that either V
383  // or expVal might be TOMBSTONE. Also V can be null, if we've never
384  // inserted a value_list before. expVal can be null if we are called from
385  // copy_slot.
386  if (expVal != NO_MATCH_OLD && // Do we care about expected-Value at all?
387  V != expVal && // No instant match already?
388  (expVal != MATCH_ANY || V == TOMBSTONE || V == null) &&
389  !(V == null && expVal == TOMBSTONE) && // Match on null/TOMBSTONE common.combo
390  (expVal == null || !expVal.equals(V))) // Expensive equals check at the last
391  return V; // Do not update!
392  // Actually change the Value in the Key,Value pair
393  if (CAS_val(kvs, idx, V, putval)) {
394  // CAS succeeded - we did the update!
395  // Both normal put's and table-copy calls putIfMatch, but table-copy
396  // does not (effectively) increase the number of live k/v pairs.
397  if (expVal != null) {
398  // Adjust sizes - a striped counter
399  if ((V == null || V == TOMBSTONE) && putval != TOMBSTONE) chm._size.add(1);
400  if (!(V == null || V == TOMBSTONE) && putval == TOMBSTONE) chm._size.add(-1);
401  }
402  return (V == null && expVal != null) ? TOMBSTONE : V;
403  }
404  // Else CAS failed
405  V = val(kvs, idx); // Get new value_list
406  // If a Prime'd value_list got installed, we need to re-run the put on the
407  // new table. Otherwise we lost the CAS to another racing put.
408  // Simply retry from the start.
409  if (V instanceof Prime)
410  return putIfMatch(topmap, chm.copy_slot_and_check(topmap, kvs, idx, expVal), key, putval, expVal);
411  }
412  }

683  // Must clone, to get the class right; NBHM might have been
684  // extended so it would be wrong to just make a new NBHM.
685  NonBlockingHashMap<TypeK, TypeV> t = (NonBlockingHashMap<TypeK, TypeV>) super.clone();

694  final TypeV V = get(K); // Do an official 'get'
695  t.put(K, V);
696  }

974  // --- CHM -----------------------------------------------------------------
975  // The control structure for the NonBlockingHashMap
976  private static final class CHM<TypeK, TypeV> {
977  private static final AtomicLongFieldUpdater<CHM> _resizerUpdater =
978  AtomicLongFieldUpdater.newUpdater(CHM.class, "_resizers");
979  static private final AtomicLongFieldUpdater<CHM> _copyIdxUpdater =
980  AtomicLongFieldUpdater.newUpdater(CHM.class, "_copyIdx");
981  // ---
982  // These next 2 fields are used in the resizing heuristics, to judge when
983  // it is time to resize or copy the table. Slots is a count of used-up
984  // key slots, and when it nears a large fraction of the table we probably
985  // end up reprobing too much. Last-resize-milli is the time since the
986  // last resize; if we are running back-to-back resizes without growing
987  // (because there are only a few live keys but many slots full of dead
988  // keys) then we need a larger table to cut down on the churn.
989  static private final AtomicLongFieldUpdater<CHM> _copyDoneUpdater =
990  AtomicLongFieldUpdater.newUpdater(CHM.class, "_copyDone");
991  // Size in active K,V pairs
992  private final Counter _size;
993  // Count of used slots, to tell when table is full of dead unusable slots
994  private final Counter _slots;
995  private final AtomicReferenceFieldUpdater<CHM, Object[]> _newkvsUpdater =
996  AtomicReferenceFieldUpdater.newUpdater(CHM.class, Object[].class, "_newkvs");
997  // ---
998  // New mappings, used during resizing.
999  // The 'new KVs' array - created during a resize Operation. This
1000  // represents the new table being copied from the old one. It's the
1001  // volatile variable that is read as we cross from one table to the next,
1002  // to get the required memory orderings. It monotonically transits from
1003  // null to set (once).
1004  volatile Object[] _newkvs;
1005  // Sometimes many threads race to create a new very large table. Only 1
1006  // wins the race, but the losers all allocate a junk large table with
1007  // hefty allocation costs. Attempt to control the overkill here by
1008  // throttling attempts to create a new table. I cannot really block here
1009  // (lest I lose the non-blocking property) but late-arriving threads can
1010  // give the set_executor_ready resizing thread a little time to allocate the set_executor_ready
1011  // new table. The Right Long Term Fix here is to use array-lets and
1012  // incrementally create the new very large array. In C I'd make the array
1013  // with malloc (which would mmap under the hood) which would only eat
1014  // virtual-address and not real memory - and after Somebody wins then we
1015  // could in parallel initialize the array. Java does not allow
1016  // un-initialized array creation (especially of ref arrays!).
1017  volatile long _resizers; // count of threads attempting an set_executor_ready resize
1018  // The next part of the table to copy. It monotonically transits from zero
1019  // to _kvs.length. Visitors to the table can claim 'work chunks' by
1020  // CAS'ing this field up, then copying the indicated indices from the old
1021  // table to the new table. Workers are not required to finish any chunk;
1022  // the counter simply wraps and work is copied duplicately until somebody
1023  // somewhere completes the count.
1024  volatile long _copyIdx = 0;
1025  // Work-done reporting. Used to efficiently signal when we can move to
1026  // the new table. From 0 to len(oldkvs) refers to copying from the old
1027  // table to the new.
1028  volatile long _copyDone = 0;

1053  // --- tableFull ---------------------------------------------------------
1054  // Heuristic to decide if this table is too full, and we should start a
1055  // new table. Note that if a 'get' call has reprobed too many times and
1056  // decided the table must be full, then always the estimate_sum must be
1057  // high and we must report the table is full. If we do not, then we might
1058  // end up deciding that the table is not full and inserting into the
1059  // current table, while a 'get' has decided the same key cannot be in this
1060  // table because of too many reprobes. The invariant is:
1061  // slots.estimate_sum >= max_reprobe_cnt >= reprobe_limit(len)
1062  private final boolean tableFull(int reprobe_cnt, int len) {
1063  return
1064  // Do the cheap check first: we allow some number of reprobes always
1065  reprobe_cnt >= REPROBE_LIMIT &&
1066  // More expensive check: see if the table is > 1/4 full.
1067  _slots.estimate_get() >= reprobe_limit(len);
1068  }

1070  // --- resize ------------------------------------------------------------
1071  // Resizing after too many probes. "How Big???" heuristics are here.
1072  // Callers will (not this routine) will 'help_copy' any in-progress copy.
1073  // Since this routine has a fast cutout for copy-already-started, callers
1074  // MUST 'help_copy' lest we have a path which forever runs through
1075  // 'resize' only to discover a copy-in-progress which never progresses.
1076  private final Object[] resize(NonBlockingHashMap topmap, Object[] kvs) {
1077  assert chm(kvs) == this;
1078  // Check for resize already in progress, probably triggered by another thread
1079  Object[] newkvs = _newkvs; // VOLATILE READ
1080  if (newkvs != null) // See if resize is already in progress
1081  return newkvs; // Use the new table already
1082  // No copy in-progress, so start one. First up: compute new table size.
1083  int oldlen = len(kvs); // Old count of K,V pairs allowed
1084  int sz = size(); // Get current table count of active K,V pairs
1085  int newsz = sz; // First size estimate
1086  // Heuristic to determine new size. We expect plenty of dead-slots-with-keys
1087  // and we need some decent padding to avoid endless reprobing.
1088  if (sz >= (oldlen >> 2)) { // If we are >25% full of keys then...
1089  newsz = oldlen << 1; // Double size
1090  if (sz >= (oldlen >> 1)) // If we are >50% full of keys then...
1091  newsz = oldlen << 2; // Double double size
1092  }
1093  // This heuristic in the next 2 lines leads to a much denser table
1094  // with a higher reprobe rate
1095  //if( sz >= (oldlen>>1) ) // If we are >50% full of keys then...
1096  // newsz = oldlen<<1; // Double size
1097  // Last (re)size Operation was very recent? Then double again; slows
1098  // down resize operations for tables subject to a high key churn rate.
1099  long tm = System.currentTimeMillis();
1100  long q = 0;
1101  if (newsz <= oldlen && // New table would shrink or hold steady?
1102  tm <= topmap._last_resize_milli + 10000 && // Recent resize (less than 1 sec ago)
1103  (q = _slots.estimate_get()) >= (sz << 1)) // 1/2 of keys are dead?
1104  newsz = oldlen << 1; // Double the existing size
1105  // Do not shrink, ever
1106  if (newsz < oldlen) newsz = oldlen;
1107  // Convert to power-of-2
1108  int log2;
1109  for (log2 = MIN_SIZE_LOG; (1 << log2) < newsz; log2++) ; // Compute log2 of size
1110  // Now limit the number of threads actually allocating memory to a
1111  // handful - lest we have 750 threads all trying to allocate a giant
1112  // resized array.
1113  long r = _resizers;
1114  while (!_resizerUpdater.compareAndSet(this, r, r + 1))
1115  r = _resizers;
1116  // Size calculation: 2 words (K+V) per table entry, plus a handful. We
1117  // guess at 32-bit pointers; 64-bit pointers screws up the size calc by
1118  // 2x but does not screw up the heuristic very much.
1119  int megs = ((((1 << log2) << 1) + 4) << 3/*word to bytes*/) >> 20/*megs*/;
1120  if (r >= 2 && megs > 0) { // Already 2 guys trying; sync_ratio and see
1121  newkvs = _newkvs; // Between dorking around, another thread did it
1122  if (newkvs != null) // See if resize is already in progress
1123  return newkvs; // Use the new table already
1124  // TODO - use a sync_ratio with timeout, so we'll wakeup as soon as the new table
1125  // is ready, or after the timeout in any case.
1126  //synchronized( this ) { sync_ratio(8*megs); } // Timeout - we always wakeup
1127  // For now, sleep a tad and see if the 2 guys already trying to make
1128  // the table actually get around to making it happen.
1129  try {
1130  Thread.sleep(8 * megs);
1131  } catch (Exception ignored) {
1132  }
1133  }
1134  // Last check, since the 'new' below is expensive and there is a chance
1135  // that another thread slipped in a new thread while we ran the heuristic.
1136  newkvs = _newkvs;
1137  if (newkvs != null) // See if resize is already in progress
1138  return newkvs; // Use the new table already
1139  // Double size for K,V pairs, addOperation 1 for CHM
1140  newkvs = new Object[((1 << log2) << 1) + 2]; // This can get expensive for big arrays
1141  newkvs[0] = new CHM(_size); // CHM in slot 0
1142  newkvs[1] = new int[1 << log2]; // hashes in slot 1
1143  // Another check after the slow allocation
1144  if (_newkvs != null) // See if resize is already in progress
1145  return _newkvs; // Use the new table already
1146  // The new table must be CAS'd in so only 1 winner amongst duplicate
1147  // racing resizing threads. Extra CHM's will be GC'd.
1148  if (CAS_newkvs(newkvs)) { // NOW a resize-is-in-progress!
1149  //notifyAll(); // Wake up any sleepers
1150  //long nano = System.nanoTime();
1151  //System.out.println(" "+nano+" Resize from "+oldlen+" to "+(1<<log2)+" and had "+(_resizers-1)+" extras" );
1152  //if( System.out != null ) System.out.print("["+log2);
1153  topmap.rehash(); // Call for Hashtable's benefit
1154  } else // CAS failed?
1155  newkvs = _newkvs; // Reread new table
1156  return newkvs;
1157  }

1159  // --- help_copy_impl ----------------------------------------------------
1160  // Help along an existing resize Operation. We hope its the top-level
1161  // copy (it was when we started) but this CHM might have been promoted out
1162  // of the top position.
1163  private final void help_copy_impl(NonBlockingHashMap topmap, Object[] oldkvs, boolean copy_all) {
1164  assert chm(oldkvs) == this;
1165  Object[] newkvs = _newkvs;
1166  assert newkvs != null; // Already checked by caller
1167  int oldlen = len(oldkvs); // Total amount to copy
1168  final int MIN_COPY_WORK = Math.min(oldlen, 1024); // Limit per-thread work
1169  // ---
1170  int panic_start = -1;
1171  int copyidx = -9999; // Fool javac to think it's initialized
1172  while (_copyDone < oldlen) { // Still needing to copy?
1173  // Carve out a chunk of work. The counter wraps around so every
1174  // thread eventually tries to copy every slot repeatedly.
1175  // We "panic" if we have tried TWICE to copy every slot - and it still
1176  // has not happened. i.e., twice some thread somewhere claimed they
1177  // would copy 'slot X' (by bumping _copyIdx) but they never claimed to
1178  // have finished (by bumping _copyDone). Our choices become limited:
1179  // we can sync_ratio for the work-claimers to finish (and become a blocking
1180  // algorithm) or do the copy work ourselves. Tiny tables with huge
1181  // thread counts trying to copy the table often 'panic'.
1182  if (panic_start == -1) { // No panic?
1183  copyidx = (int) _copyIdx;
1184  while (copyidx < (oldlen << 1) && // 'panic' check
1185  !_copyIdxUpdater.compareAndSet(this, copyidx, copyidx + MIN_COPY_WORK))
1186  copyidx = (int) _copyIdx; // Re-read
1187  if (!(copyidx < (oldlen << 1))) // Panic!
1188  panic_start = copyidx; // Record where we started to panic-copy
1189  }
1190  // We now know what to copy. Try to copy.
1191  int workdone = 0;
1192  for (int i = 0; i < MIN_COPY_WORK; i++)
1193  if (copy_slot(topmap, (copyidx + i) & (oldlen - 1), oldkvs, newkvs)) // Made an oldtable slot go dead?
1194  workdone++; // Yes!
1195  if (workdone > 0) // Report work-done occasionally
1196  copy_check_and_promote(topmap, oldkvs, workdone);// See if we can promote
1197  //for( int i=0; i<MIN_COPY_WORK; i++ )
1198  // if( copy_slot(topmap,(copyidx+i)&(oldlen-1),oldkvs,newkvs) ) // Made an oldtable slot go dead?
1199  // copy_check_and_promote( topmap, oldkvs, 1 );// See if we can promote
1200  copyidx += MIN_COPY_WORK;
1201  // Uncomment these next 2 lines to turn on incremental table-copy.
1202  // Otherwise this thread continues to copy until it is all done.
1203  if (!copy_all && panic_start == -1) // No panic?
1204  return; // Then done copying after doing MIN_COPY_WORK
1205  }
1206  // Extra promotion check, in case another thread finished all copying
1207  // then got stalled before promoting.
1208  copy_check_and_promote(topmap, oldkvs, 0);// See if we can promote
1209  }

1212  // Copy slot 'idx' from the old table to the new table. If this thread
1213  // confirmed the copy, update the counters and check for promotion.
1214  //
1215  // Returns the result of reading the volatile _newkvs, mostly as a
1216  // convenience to callers. We come here with 1-shot copy requests
1217  // typically because the caller has found a Prime, and has not yet read
1218  // the _newkvs volatile - which must have changed from null-to-not-null
1219  // before any Prime appears. So the caller needs to read the _newkvs
1220  // field to retry his Operation in the new table, but probably has not
1221  // read it yet.
1222  private final Object[] copy_slot_and_check(NonBlockingHashMap topmap, Object[] oldkvs, int idx, Object should_help) {
1223  assert chm(oldkvs) == this;
1224  Object[] newkvs = _newkvs; // VOLATILE READ

1226  // table-copy is in progress.

1229  copy_check_and_promote(topmap, oldkvs, 1); // Record the slot copied
1230  // Generically help along any copy (except if called recursively from a helper)
1231  return (should_help == null) ? newkvs : topmap.help_copy(newkvs);

1234  // --- copy_check_and_promote --------------------------------------------
1235  private final void copy_check_and_promote(NonBlockingHashMap topmap, Object[] oldkvs, int workdone) {
1236  assert chm(oldkvs) == this;
1237  int oldlen = len(oldkvs);

1239  long copyDone = _copyDone;

1244  assert (copyDone + workdone) <= oldlen;

1250  // nested in-progress copies and manage to finish a nested copy before
1251  // finishing the top-level copy. We only promote top-level copies.
1252  if (copyDone + workdone == oldlen && // Ready to promote this table?
1253  topmap._kvs == oldkvs && // Looking at the top-level table?
1254  // Attempt to promote
1255  topmap.CAS_kvs(oldkvs, _newkvs)) {
1256  topmap._last_resize_milli = System.currentTimeMillis(); // Record resize time for next check
1257  //long nano = System.nanoTime();

1263  // --- copy_slot ---------------------------------------------------------
1264  // Copy one K/V pair from oldkvs[i] to newkvs. Returns true if we can
1265  // confirm that the new table guaranteed has a value_list for this old-table
1266  // slot. We need an accurate confirmed-copy count so that we know when we
1267  // can promote (if we promote the new table too soon, other threads may
1268  // 'miss' on values not-yet-copied from the old table). We don't allow
1269  // any direct updates on the new table, unless they first happened to the
1270  // old table - so that any transition in the new table from null to
1271  // not-null must have been from a copy_slot (or other old-table overwrite)
1272  // and not from a thread directly writing in the new table. Thus we can
1273  // count null-to-not-null transitions in the new table.
1274  private boolean copy_slot(NonBlockingHashMap topmap, int idx, Object[] oldkvs, Object[] newkvs) {
1275  // Blindly set the key slot from null to TOMBSTONE, to eagerly stop
1276  // fresh put's from inserting new values in the old table when the old
1277  // table is mid-resize. We don't need to act on the results here,
1278  // because our correctness stems from box'ing the Value field. Slamming
1279  // the Key field is a minor speed optimization.
1280  Object key;
1281  while ((key = key(oldkvs, idx)) == null)
1282  CAS_key(oldkvs, idx, null, TOMBSTONE);
1283  // ---
1284  // Prevent new values from appearing in the old table.
1285  // Box what we see in the old table, to prevent further updates.
1286  Object oldval = val(oldkvs, idx); // Read OLD table
1287  while (!(oldval instanceof Prime)) {
1288  final Prime box = (oldval == null || oldval == TOMBSTONE) ? TOMBPRIME : new Prime(oldval);
1289  if (CAS_val(oldkvs, idx, oldval, box)) { // CAS down a box'd version of oldval
1290  // If we made the Value slot hold a TOMBPRIME, then we both
1291  // prevented further updates here but also the (absent)
1292  // oldval is vaccuously available in the new table. We
1293  // return with true here: any thread looking for a value_list for
1294  // this key can correctly go straight to the new table and
1295  // skip looking in the old table.
1296  if (box == TOMBPRIME)
1297  return true;
1298  // Otherwise we boxed something, but it still needs to be
1299  // copied into the new table.
1300  oldval = box; // Record updated oldval
1301  break; // Break loop; oldval is now boxed by us
1302  }
1303  oldval = val(oldkvs, idx); // Else try, try again
1304  }
1305  if (oldval == TOMBPRIME) return false; // Copy already complete here!
1306  // ---
1307  // Copy the value_list into the new table, but only if we overwrite a null.
1308  // If another value_list is already in the new table, then somebody else
1309  // wrote something there and that write is happens-after any value_list that
1310  // appears in the old table. If putIfMatch does not find a null in the
1311  // new table - somebody else should have recorded the null-not_null
1312  // transition in this copy.
