38  private static Set<String> getResourcesOnBootClasspath() {
39  final Logger logger = LoggerFactory.getLogger(BootClassPath.class);
40  final Set<String> resources = new HashSet<>();
41  final String bootClassPath = System.getProperty("sun.boot.class.path", "");
42  logger.trace("Boot class-path is: {}", bootClassPath);
43 
44  final String pathSeparator = System.getProperty("path.separator");
45  logger.trace("Path separator is: '{}'", pathSeparator);
46 
47  final String[] pathElements = bootClassPath.split(pathSeparator);
48 
49  for (final String pathElement : pathElements) {
50  resources.addAll(findResources(Paths.get(pathElement), logger));
51  }
52 
53  return resources;
54  }

56  private static Set<String> findResources(final Path path, final Logger logger) {
57  if (!Files.exists(path)) {
58  return Collections.emptySet();
59  }
60 
61  if (Files.isDirectory(path)) {
62  return findResourcesInDirectory(path, logger);
63  }
64 
65  return findResourcesInJar(path, logger);
66  }

68  private static Set<String> findResourcesInJar(final Path path, final Logger logger) {
69  final Set<String> jarResources = new HashSet<>();
70  try {
71  final JarFile jarFile = new JarFile(path.toFile());
72  final Enumeration<JarEntry> entries = jarFile.entries();
73  while (entries.hasMoreElements()) {
74  final JarEntry jarEntry = entries.nextElement();
75  if (jarEntry.getName().endsWith(".class")) {
76  jarResources.add(jarEntry.getName());
77  }
78  }
79  } catch (IOException e) {
80  logger.warn("Not a jar file: {}", path);
81  }
82 
83  return jarResources;
84  }

86  private static Set<String> findResourcesInDirectory(final Path path, final Logger logger) {
87  final Set<String> dirResources = new HashSet<>();
88  try {
89  Files.walkFileTree(path, new SimpleFileVisitor<Path>() {
90  @Override
91  public FileVisitResult visitFile(final Path file, final BasicFileAttributes attrs) throws IOException {
92  if (file.getFileName().toString().endsWith(".class")) {
93  dirResources.add(path.relativize(file).toString());
94  }
95  return super.visitFile(file, attrs);
96  }
97  });
98  } catch (IOException e) {
99  logger.warn("Error walking dir: " + path, e);
100  }
101 
102  return dirResources;
103  }
