90  BitSet affinity2 = getAffinity0();

98  final CLibrary lib = CLibrary.INSTANCE;
99  final LongByReference cpuset1 = new LongByReference(0);
100  final LongByReference cpuset2 = new LongByReference(0);
101  try {
102 
103  final int ret = lib.GetProcessAffinityMask(handle(-1), cpuset1, cpuset2);
104  // Successful result is positive, according to the docs
105  // https://msdn.microsoft.com/en-us/library/windows/desktop/ms683213%28v=vs.85%29.aspx

107  throw new IllegalStateException("GetProcessAffinityMask(( -1 ), &(" + cpuset1 + "), &(" + cpuset2 + ") ) return " + ret);
