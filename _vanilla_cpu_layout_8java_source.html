39  VanillaCpuLayout(List<CpuInfo> cpuDetails) {
40  this.cpuDetails = cpuDetails;
41  SortedSet<Integer> sockets = new TreeSet<>(),
42  cores = new TreeSet<>(),
43  threads = new TreeSet<>();
44  for (CpuInfo cpuDetail : cpuDetails) {
45  sockets.add(cpuDetail.socketId);
46  cores.add((cpuDetail.socketId << 16) + cpuDetail.coreId);
47  threads.add(cpuDetail.threadId);
48  }
49  this.sockets = sockets.size();
50  this.coresPerSocket = cores.size() / sockets.size();
51  this.threadsPerCore = threads.size();
52  if (cpuDetails.size() != sockets() * coresPerSocket() * threadsPerCore()) {
53  StringBuilder error = new StringBuilder();
54  error.append("cpuDetails.size= ").append(cpuDetails.size())
55  .append(" != sockets: ").append(sockets())
56  .append(" * coresPerSocket: ").append(coresPerSocket())
57  .append(" * threadsPerCore: ").append(threadsPerCore()).append('\n');
58  for (CpuInfo detail : cpuDetails) {
59  error.append(detail).append('\n');
60  }
61  LoggerFactory.getLogger(VanillaCpuLayout.class).warn(error.toString());
62  }
63  }

110  public static VanillaCpuLayout fromCpuInfo(InputStream is) throws IOException {
111  BufferedReader br = new BufferedReader(new InputStreamReader(is, "UTF-8"));
112  String line;
113  List<CpuInfo> cpuDetails = new ArrayList<>();
114  CpuInfo details = new CpuInfo();
115  Map<String, Integer> threadCount = new LinkedHashMap<>();
116 
117  while ((line = br.readLine()) != null) {
118  if (line.trim().isEmpty()) {
119  String key = details.socketId + "," + details.coreId;
120  Integer count = threadCount.get(key);
121  if (count == null)
122  threadCount.put(key, count = 1);
123  else
124  threadCount.put(key, count += 1);
125  details.threadId = count - 1;
126  cpuDetails.add(details);
127  details = new CpuInfo();
128  details.coreId = cpuDetails.size();
129  continue;
130  }
131  String[] words = line.split("\\s*:\\s*", 2);
132  if (words[0].equals("physical id"))
133  details.socketId = parseInt(words[1]);
134  else if (words[0].equals("core id"))
135  details.coreId = parseInt(words[1]);
136  }
137  return new VanillaCpuLayout(cpuDetails);
138  }
