165  private static double runTopologyLocally(Topology topology, Configuration conf) throws InterruptedException {
166  TopologySubmitter submitter = new TopologySubmitter();
167  try {
168  final_topology = submitter.submitTopology(topology, conf);
169  } catch (UnhandledCaseException e) {
170  e.printStackTrace();
171  }
172  executorThread sinkThread = submitter.getOM().getEM().getSinkThread();
173  long start = System.currentTimeMillis();
174  sinkThread.join((long) (30 * 1E3 * 60));//sync_ratio for sink thread to stop. Maximally sync_ratio for 10 mins
175  long time_elapsed = (long) ((System.currentTimeMillis() - start) / 1E3 / 60);//in mins
176  if (time_elapsed > 20) {
177  if (enable_log) log.info("Program error, exist...");
178  System.exit(-1);
179  }
180 
181  submitter.getOM().join();
182  submitter.getOM().getEM().exist();
183  if (sinkThread.running) {
184  if (enable_log) log.info("The application fails to stop normally, exist...");
185  return -1;
186  } else {
187  if (enable_app_combo) {
188  return SINK_CONTROL.getInstance().throughput;
189  } else {
190  TopologyComponent sink = submitter.getOM().g.getSink().operator;
191  double sum = 0;
192  int cnt = 0;
193  for (ExecutionNode e : sink.getExecutorList()) {
194  double results = e.op.getResults();
195  if (results != 0) {
196  sum += results;
197  } else {
198  sum += sum / cnt;
199  }
200  cnt++;
201  }
202  return sum;
203  }
